<script>
  const speedyIcon = document.getElementById('speedyIcon');
  const metalIcon = document.getElementById('metalIcon');
  const oliviaIcon = document.getElementById('oliviaIcon');

  const speedyBrowser = document.getElementById('speedyBrowser');
  const metalLauncher = document.getElementById('metalLauncher');
  const oliviaApp = document.getElementById('oliviaApp');
  const oliviaAdPopup = document.getElementById('oliviaAdPopup');

  const yesBtn = document.querySelector('.yes');
  const noBtn = document.querySelector('.no');

  const searchBox = document.getElementById('searchBox');
  const webView = document.getElementById('webView');

  let searchCount = 0;
  let oliviaInstalled = false;

  speedyIcon.onclick = () => {
    speedyBrowser.style.display = 'block';
    metalLauncher.style.display = 'none';
    oliviaApp.style.display = 'none';
  };

  metalIcon.onclick = () => {
    speedyBrowser.style.display = 'none';
    metalLauncher.style.display = 'block';
    oliviaApp.style.display = 'none';
  };

  oliviaIcon.onclick = () => {
    if (oliviaInstalled) {
      oliviaApp.style.display = 'block';
      speedyBrowser.style.display = 'none';
      metalLauncher.style.display = 'none';
    } else {
      oliviaAdPopup.style.display = 'flex';
    }
  };

  yesBtn.onclick = () => {
    oliviaInstalled = true;
    oliviaAdPopup.style.display = 'none';
    oliviaApp.style.display = 'block';
  };

  noBtn.onclick = () => {
    oliviaAdPopup.style.display = 'none';
    alert("Olivia is installing anyway...");
    oliviaInstalled = true;
    setTimeout(() => {
      oliviaApp.style.display = 'block';
    }, 1000);
  };

  searchBox.addEventListener('keydown', (e) => {
    if (e.key === 'Enter') {
      const val = searchBox.value;
      searchCount++;
      webView.innerHTML += `<p>You searched for: <strong>${val}</strong></p>`;
      if (searchCount >= 5) {
        alert("Warning: Olivia is watching you...");
      }
      searchBox.value = '';
    }
  });
</script>
